<script>

    let correctAnswer = 65;

    let totalEnergy = 100;

    let runningEnergy = 65;

   

    function newProblem() {

        // 공급된 휘발유 에너지 (60-100 MJ)

        totalEnergy = Math.floor(Math.random() * 41) + 60;

       

        // 배기 가스 (전체의 30-50%)

        const exhaustPercent = Math.floor(Math.random() * 21) + 30;

        const exhaust = Math.round(totalEnergy * exhaustPercent / 100);

       

        // 달리는 데 사용한 에너지

        runningEnergy = totalEnergy - exhaust;

       

        // 에너지 효율 계산

        correctAnswer = Math.round((runningEnergy / totalEnergy) * 100);

       

        document.getElementById('totalEnergy').textContent = totalEnergy;

        document.getElementById('running').textContent = '❓';

        document.getElementById('exhaust').textContent = exhaust;

        document.getElementById('answer').value = '';

        document.getElementById('result').style.display = 'none';

    }

   

    function checkAnswer() {

        const userAnswer = parseInt(document.getElementById('answer').value);

        const resultDiv = document.getElementById('result');

       

        if (isNaN(userAnswer)) {

            resultDiv.className = 'result incorrect';

            resultDiv.innerHTML = '<span class="emoji">⚠️</span>답을 입력해주세요!';

            resultDiv.style.display = 'block';

            return;

        }

       

        if (userAnswer === correctAnswer) {

            resultDiv.className = 'result correct';

            resultDiv.innerHTML = `

                <span class="emoji">🎉</span>정답입니다!<br><br>

                <strong>풀이:</strong><br>

                달리는 데 사용한 에너지 = ${totalEnergy} - ${document.getElementById('exhaust').textContent} = ${runningEnergy} MJ<br><br>

                엔진의 에너지 효율 = (${runningEnergy} ÷ ${totalEnergy}) × 100<br>

                <strong>= ${correctAnswer}%</strong>

            `;

            document.getElementById('running').textContent = runningEnergy;

        } else {

            resultDiv.className = 'result incorrect';

            resultDiv.innerHTML = `

                <span class="emoji">❌</span>틀렸습니다.<br><br>

                힌트:<br>

                1. 먼저 달리는 데 사용한 에너지를 구하세요.<br>

                (공급된 에너지 - 배기 가스)<br>

                2. 에너지 효율 = (달리는 데 사용한 에너지 ÷ 공급된 에너지) × 100

            `;

        }

       

        resultDiv.style.display = 'block';

    }

   

    document.getElementById('answer').addEventListener('keypress', function(e) {

        if (e.key === 'Enter') {

            checkAnswer();

        }

    });

   

    newProblem();

</script>
